\begin{tikzpicture}
    \draw (0,0) -- (4,0);
    \draw[dash pattern=on 4pt off 4pt] (0,-1) -- (4,-1);
    \draw[dash pattern=on 2pt off 6pt on 6pt off 2pt] (0,-2) -- (4,-2);
    \draw[dash pattern=on 4pt off 4pt, dash phase=4pt] (0,-3) -- (4,-3);
    \draw[dash pattern=on 2pt off 6pt on 6pt off 2pt, dash phase=4pt] (0,-4) -- (4,-4);
\end{tikzpicture}
